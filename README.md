## Install

```bash
npm install --save icedui
```

## Usage

```jsx
import React, { Component } from 'react'

import MyComponent from 'icedui'
import 'icedui/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
